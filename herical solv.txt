using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp13
{
    class A
    {
        public string name;

    }
    class B:A
    {
        public int id;
    }
    class C : A
    {
        public int age;

    }
    internal class Program
    {
        static void Main(string[] args)
        {
            
            B b = new B();
            b.id = 1;
            b.name = "naman";
            C c = new C();
            c.age = 23;
            c.name = "naman";

            Console.WriteLine(b.name+"  "+b.id+"  "+c.age);
            Console.ReadLine(); 
        }
    }
}
